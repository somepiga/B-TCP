104template <typename PollingType>
105Result EventLoop<PollingType>::wait_next_event(const int timeout_ms) {
106 // 先处理非文件描述符的规则
107 {
108 for (auto it = _non_fd_rules.begin(); it != _non_fd_rules.end();) {
109 auto& this_rule = **it;
110 bool rule_fired = false;
111
112 if (this_rule.cancel_requested) {
113 it = _non_fd_rules.erase(it);
114 continue;
115 }
116
117 uint8_t iterations = 0;
118 while (this_rule.interest()) {
119 if (iterations++ >= 128) {
120 throw std::runtime_error(
121 "EventLoop: busy wait detected: rule \"" +
122 _rule_categories.at(this_rule.category_id).name +
123 "\" is still interested after " + std::to_string(iterations) +
124 " iterations");
125 }
126
127 rule_fired = true;
128 this_rule.callback();
129 }
130
131 if (rule_fired) {
132 return Result::Success; /* 每次迭代只处理一条规则 */
133 }
134
135 ++it;
136 }
137 }
138
139 // 处理文件描述符的规则
140 bool something_to_poll = false;
141
142 for (auto it = _fd_rules.begin(); it != _fd_rules.end();) {
143 auto& this_rule_addr = *it;
144 auto& this_rule = **it;
145
146 if (this_rule.cancel_requested) {
147 it = _fd_rules.erase(it);
148 continue;
149 }
150
151 if (this_rule.direction == Direction::In && this_rule.fd.eof()) {
152 // no more reading on this rule, it's reached eof
153 this_rule.cancel();
154 it = _fd_rules.erase(it);
155 continue;
156 }
157
158 if (this_rule.fd.closed()) {
159 this_rule.cancel();
160 it = _fd_rules.erase(it);
161 continue;
162 }
163
164 if (this_rule.interest()) {
165 _polling.insert_rule(this_rule_addr);
166 something_to_poll = true;
167 }
168 ++it;
169 }
170
171 // 如果没有什么可轮询的则退出
172 if (not something_to_poll) {
173 _polling.clear();
174 return Result::Exit;
175 }
176
177 _polling.insert_to_epoll();
178
179 if (-1 == _polling.b_epoll_wait(timeout_ms)) {
180 _polling.clear();
181 return Result::Timeout;
182 }
183
184 _polling.call(_fd_rules);
185
186 _polling.clear();
187
188 return Result::Success;
189}
