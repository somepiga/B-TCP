109template <typename PollingType>
110Result EventLoop<PollingType>::wait_next_event(const int timeout_ms) {
111 // first, handle the non-file-descriptor-related rules
112 {
113 for (auto it = _non_fd_rules.begin(); it != _non_fd_rules.end();) {
114 auto& this_rule = **it;
115 bool rule_fired = false;
116
117 if (this_rule.cancel_requested) {
118 it = _non_fd_rules.erase(it);
119 continue;
120 }
121
122 uint8_t iterations = 0;
123 while (this_rule.interest()) {
124 if (iterations++ >= 128) {
125 throw runtime_error("EventLoop: busy wait detected: rule \"" +
126 _rule_categories.at(this_rule.category_id).name +
127 "\" is still interested after " +
128 to_string(iterations) + " iterations");
129 }
130
131 rule_fired = true;
132 this_rule.callback();
133 }
134
135 if (rule_fired) {
136 return Result::Success; /* only serve one rule on each iteration */
137 }
138
139 ++it;
140 }
141 }
142
143 // now the file-descriptor-related rules. poll any "interested" file
144 // descriptors
145 vector<pollfd> pollfds{};
146 pollfds.reserve(_fd_rules.size());
147 bool something_to_poll = false;
148
149 // set up the pollfd for each rule
150 for (auto it = _fd_rules.begin();
151 it !=
152 _fd_rules.end();) { // NOTE: it gets erased or incremented in loop body
153 auto& this_rule = **it;
154
155 if (this_rule.cancel_requested) {
156 // this_rule.cancel();
157 // if rule is cancelled externally, no need to call the cancellation
158 // callback this makes it easier to cancel rules and delete captured
159 // objects right away
160 it = _fd_rules.erase(it);
161 continue;
162 }
163
164 if (this_rule.direction == Direction::In && this_rule.fd.eof()) {
165 // no more reading on this rule, it's reached eof
166 this_rule.cancel();
167 it = _fd_rules.erase(it);
168 continue;
169 }
170
171 if (this_rule.fd.closed()) {
172 this_rule.cancel();
173 it = _fd_rules.erase(it);
174 continue;
175 }
176
177 if (this_rule.interest()) {
178 pollfds.push_back({this_rule.fd.fd_num(),
179 static_cast<int16_t>(this_rule.direction), 0});
180 something_to_poll = true;
181 } else {
182 pollfds.push_back({this_rule.fd.fd_num(), 0,
183 0}); // placeholder --- we still want errors
184 }
185 ++it;
186 }
187
188 // quit if there is nothing left to poll
189 if (not something_to_poll) {
190 return Result::Exit;
191 }
192
193 // call poll -- wait until one of the fds satisfies one of the rules
194 // (writeable/readable)
195 if (0 == CheckSystemCall(
196 "poll", ::poll(pollfds.data(), pollfds.size(), timeout_ms))) {
197 return Result::Timeout;
198 }
199
200 // go through the poll results
201 for (auto [it, idx] = make_pair(_fd_rules.begin(), static_cast<size_t>(0));
202 it != _fd_rules.end(); ++idx) {
203 const auto& this_pollfd = pollfds.at(idx);
204 auto& this_rule = **it;
205
206 const auto poll_error =
207 static_cast<bool>(this_pollfd.revents & (POLLERR | POLLNVAL));
208 if (poll_error) {
209 /* recoverable error? */
210 if (not static_cast<bool>(this_pollfd.revents & POLLNVAL)) {
211 if (this_rule.recover()) {
212 ++it;
213 continue;
214 }
215 }
216
217 /* see if fd is a socket */
218 int socket_error = 0;
219 socklen_t optlen = sizeof(socket_error);
220 const int ret = getsockopt(this_rule.fd.fd_num(), SOL_SOCKET, SO_ERROR,
221 &socket_error, &optlen);
222 if (ret == -1 and errno == ENOTSOCK) {
223 cerr << "error on polled file descriptor for rule \""
224 << _rule_categories.at(this_rule.category_id).name << "\"\n";
225 } else if (ret == -1) {
226 throw unix_error("getsockopt");
227 } else if (optlen != sizeof(socket_error)) {
228 throw runtime_error("unexpected length from getsockopt: " +
229 to_string(optlen));
230 } else if (socket_error) {
231 cerr << "error on polled socket for rule \""
232 << _rule_categories.at(this_rule.category_id).name
233 << "\": " << strerror(socket_error) << "\n";
234 }
235
236 this_rule.cancel();
237 it = _fd_rules.erase(it);
238 continue;
239 }
240
241 const auto poll_ready =
242 static_cast<bool>(this_pollfd.revents & this_pollfd.events);
243 const auto poll_hup = static_cast<bool>(this_pollfd.revents & POLLHUP);
244 if (poll_hup && ((this_pollfd.events && !poll_ready) or
245 (this_rule.direction == Direction::Out))) {
246 // if we asked for the status, and the _only_ condition was a hangup, this
247 // FD is defunct:
248 // - if it was POLLIN and nothing is readable, no more will ever be
249 // readable
250 // - if it was POLLOUT, it will not be writable again
251 // additionally, consider FD defunct if rule will only query for
252 // Direction::Out
253 this_rule.cancel();
254 it = _fd_rules.erase(it);
255 continue;
256 }
257
258 if (poll_ready) {
259 // we only want to call callback if revents includes the event we asked
260 // for
261 const auto count_before = this_rule.service_count();
262 this_rule.callback();
263
264 if (count_before == this_rule.service_count() and
265 (not this_rule.fd.closed()) and this_rule.interest()) {
266 throw runtime_error("EventLoop: busy wait detected: rule \"" +
267 _rule_categories.at(this_rule.category_id).name +
268 "\" did not read/write fd and is still interested");
269 }
270
271 return Result::Success; /* only serve one rule on each iteration */
272 }
273
274 ++it; // if we got here, it means we didn't call _fd_rules.erase()
275 }
276
277 return Result::Success;
278}
